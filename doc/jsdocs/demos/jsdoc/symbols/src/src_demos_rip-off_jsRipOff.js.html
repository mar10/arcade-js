134 var RipOffGame = ArcadeJS.extend({
135 	init: function(canvas, customOpts) {
136 		// Init ArcadeJS
137 		var opts = $.extend({
138 			name: "jsRipOff",
139 			fps: 30,
140 			twoPlayer: false
141 		}, customOpts);
142 		this._super(canvas, opts);
143 
144 		var i;
145 		// Copy selected options as object attributes
146 		ArcadeJS.extendAttributes(this, opts, "twoPlayer");
147 
148 		// Set the scene
149 		this.setViewport(0, 0, 640, 480, "extend");
150 
151 		// --- Status data -----------------------------------------------------
152 		this.wave = 0;
153 		this.score = 0;
154 		this.score2 = 0;
155 		this.gracePeriod = 2; // seconds
156 
157 		// --- Cache sounds ----------------------------------------------------
158 		this.gunSound = new AudioJS(["fire.mp3", "fire.oga"]);
159 		this.explosionSound = new AudioJS(["damage.mp3", "damage.oga"]);
160 
161 		this.attackStyle = "orange";
162 
163 		this.player1 = this.addObject(new Tank({
164 			id: "player1",
165 			pos: new Point2(840, 100),
166 			homePos: new Point2(540, 100),
167 			color: "skyblue"
168 		}));
169 		this.player2 = this.twoPlayer ? this.addObject(new Tank({
170 			id: "player2",
171 			pos: new Point2(-200, 100),
172 			homePos: new Point2(100, 100),
173 			color: "yellowgreen"
174 		})) : null;
175 		// Seed 8 canisters in the center
176 		for(i=0; i<8; i++){
177 			var pos = new Point2(LinaJS.random(250, 390), LinaJS.random(190, 290));
178 			this.addObject(new Canister({pos: pos}));
179 		}
180 		//
181 		this._startWave();
182 		this.player1.restart();
183 		if(this.player2){
184 			this.player2.restart();
185 		}
186 
187 		// Start render loop
188 		this.startLoop();
189 	},
190 
191 	_startWave: function(){
192 		var i,
193 			maxBandits = 3,
194 			canisters = this.getObjectsByType("canister"),
195 			forceAttack = false,
196 			banditHome = new Point2(LinaJS.random(0, 640), 500);
197 
198 		this.wave += 1;
199 
200 		// Remove old bandits and bullets
201 		this.visitObjects(function(obj){
202 			obj.die();
203 		}, "bandit bullet")
204 
205 		// Create bandits and assign targets
206 		var idx = (this.wave - 1) % 6,
207 			lap = Math.floor((this.wave - 1) / 6),
208 			banditDef = banditsDefs[idx];
209 
210 		for(i=0; i<maxBandits + lap; i++){
211 			var bandit = this.addObject(new Bandit({
212 				pos: banditHome,
213 				score: banditDef.score,
214 				maxSpeed: banditDef.maxSpeed,
215 				turnRate: banditDef.turnRate,
216 				accel: banditDef.accel,
217 				decel: banditDef.decel,
218 				attackRange: banditDef.attackRange,
219 				fireRate: banditDef.fireRate,
220 				pg: banditDef.pg.copy()
221 				}));
222 
223 //			if(forceAttack && i === 0){
224 //				bandit.target = this.player1;
225 //			}else if(i < canisters.length){
226 //				bandit.target = canisters[i];
227 //			}else{
228 //				bandit.target = this.player1;
229 //			}
230 		}
231 		this._assignTargets();
232 
233 		this.setActivity("prepare");
234 		this.later(3, function(){
235 			this.setActivity("run");
236 		});
237 	},
238 	/**
239 	 * Check if targets are still valid, and re-assign as necessary
240 	 */
241 	_assignTargets: function(){
242 		var i, canister, bandit,
243 			bandits = this.getObjectsByType("bandit"),
244 			canisters = this.getObjectsByType("canister"),
245 			attachedCanisterMap = {};
246 		// Collect attached/targeted canisters and remove invalid targets
247 		for(i=0; i<bandits.length; i++){
248 			bandit = bandits[i];
249 			if(bandit.canister){
250 				attachedCanisterMap[bandit.canister.id] = bandit;
251 			}else if(bandit.target && bandit.target.type == "canister"){
252 				attachedCanisterMap[bandit.target.id] = bandit;
253 			}
254 			if(bandit.target && bandit.target.hidden){
255 				bandit.target = null;
256 			}
257 		}
258 		// Attache free canisters to lazy bandits
259 		var nextIdx = 0;
260 		for(i=0; i<bandits.length; i++){
261 			bandit = bandits[i];
262 			if(!bandit.target){
263 				canister = null;
264 				while(!canister && nextIdx < canisters.length){
265 					canister = canisters[nextIdx];
266 					if(attachedCanisterMap[canister.id]){
267 						nextIdx++;
268 						canister = null;
269 					}
270 				}
271 				if(canister){
272 					bandit.target = canister;
273 					attachedCanisterMap[canister.id] = bandit;
274 				}else{
275 					if(bandit.game.player2 && (i % 2)){
276 						bandit.target = bandit.game.player2;
277 					}else{
278 						bandit.target = bandit.game.player1;
279 					}
280 				}
281 				bandit.game.debug("Assign " + bandit.target + " to " + bandit);
282 			}
283 		}
284 	},
285 	postStep: function(){
286 		var bandits = this.getObjectsByType("bandit");
287 //		this.debug("postStep: " + bandits);
288 		if( bandits.length < 1){
289 			var canisters = this.getObjectsByType("canister");
290 			if(canisters.length){
291 				this._startWave();
292 			}else if(this.getActivity() != "over"){
293 				this.setActivity("over");
294 				var popUp = new HtmlOverlay({
295 					game: this,
296 					html: "Game Over.",
297 					onClick: function(e){
298 						window.location.reload();
299 					}
300 				});
301 			}
302 		}
303 	},
304 	preDraw: function(ctx){
305 		ctx.save();
306 		var yOfs = this.canvasArea.y;
307 		// Display score
308 		ctx.save();
309 		ctx.font = "16px sans-serif";
310 		ctx.fillStyle = this.player1.color;
311 		ctx.fillScreenText("Player 1: " + this.score, 10, 17 + yOfs);
312 		if(this.twoPlayer){
313 			ctx.fillStyle = this.player2.color;
314 			ctx.fillScreenText("Player 2: " + this.score2, 10, 34 + yOfs);
315 		}
316 		ctx.restore();
317 
318 		if(this.isActivity("prepare")){
319 			ctx.font = "30px sans-serif";
320 			ctx.strokeScreenText("Wave " + this.wave + "...", 0, 0);
321 		}else if(this.isActivity("over")){
322 			ctx.font = "30px sans-serif";
323 			ctx.strokeScreenText("Game over (hit [F5])", 0, 0);
324 		}
325 		if(!this.opts.debug.showMouse){
326 			ctx.font = "10px sans-serif";
327 			ctx.fillScreenText("Copyright (c) 2011 Martin Wendt - Made with ArcadeJS", 0, -1);
328 		}
329 		// done
330 		ctx.restore();
331 	},
332 	// --- end of class
333 	__lastentry: undefined
334 });

338 var Bullet = Movable.extend({
339 	init: function(opts) {
340 		this._super("bullet", $.extend({
341 			source: null,
342 			ttl: 1, // live max. 1 second
343 			clipModeX: "die",
344 			clipModeY: "die"
345 		}, opts));
346 		// Copy selected options as object attributes
347 		ArcadeJS.extendAttributes(this, opts, "source ttl");
348 		if(this.ttl > 0){
349 			this.later(this.ttl, function(){
350 				this.die();
351 			});
352 		}
353 	},
354 	getBoundingCircle: function() {
355 		return new Circle2(this.pos, 0.1);
356 	},
357 	step: function() {
358 		// Check if this bullet hit a player or bandit
359 		var list = this.game.getObjectsByType("tank bandit");
360 		for(var i=0, l=list.length; i<l; i++) {
361 			var obj = list[i];
362 			if(this.source === obj || !this.game.preCheckCollision(this, obj)){
363 				continue;
364 			}
365 			// Pre-check is exact enough for our purpose...
366 			this.die();
367 			obj.hitBy(this);
368 			break;
369 		}
370 	},
371 	render: function(ctx) {
372 		ctx.fillRect(0, 0, 3, 3);
373 	},
374 	// --- end of class
375 	__lastentry: undefined
376 });

400 var Tank = Movable.extend({
401 	init: function(opts) {
402 		this._super("tank", $.extend({
403 			clipModeX: "stop",
404 			clipModeY: "stop",
405 			color: "white"
406 		}, opts));
407 		// Copy selected options as object attributes
408 		ArcadeJS.extendAttributes(this, opts, "color");
409 		this.pg = pgTank1.copy();
410 //		this.pgHull = this.pg.getConvexHull();
411 		// Start here
412 		this.homePos = opts.homePos.copy();
413 		// Drive from here to homePos after a hit (recover mode).
414 		this.recoverPos = opts.pos.copy();
415 //		alert("init: " + this.id + " home:" + this.homePos + ", recover:" + this.recoverPos);
416 		this.fireRate = 330; // ms
417 		this.fireRange = 1.3; // sec
418 	},
419 	getBoundingCircle: function() {
420 		return new Circle2(this.pos, 18);
421 	},
422 	applyControls: function(){
423 		var maxSpeed = 150,
424 			accel = 150,
425 			decel = 100,
426 			turnRate = 90 * LinaJS.D2R,
427 			game = this.game;
428 
429 		// In recover mode (after it was hit), the tank drives back to it's home
430 		if(this.isActivity("recover") ){
431 			if(this.driveToPosition(game.frameDuration, this.homePos,
432 					10, maxSpeed, turnRate, accel, decel)){
433 				this.setActivity("idle");
434 				this.clipModeX = "stop";
435 			};
436 			return;
437 		}
438 		// --- Handle key controls ---
439 		if(this.id == "player1"){
440 			// Player 1 is keyboard controlled
441 			if(game.isKeyDown(32)){ // [space]
442 				this.fire();
443 			}
444 			if(game.isKeyDown(38)){ // [up]
445 				var force = LinaJS.polarToVec(this.orientation + 90*LinaJS.D2R, accel * game.frameDuration);
446 				this.velocity.accelerate(force, maxSpeed);
447 			}else{
448 				this.velocity.accelerate(-decel * game.frameDuration);
449 			}
450 			if(this.game.isKeyDown(37)){ // [left]
451 				this.orientation += turnRate * game.frameDuration;
452 				this.velocity.setAngle(this.orientation + 90*LinaJS.D2R);
453 			}else if(this.game.isKeyDown(39)){ // [right]
454 				this.orientation -= turnRate * game.frameDuration;
455 				this.velocity.setAngle(this.orientation + 90*LinaJS.D2R);
456 			}
457 		}
458 		if(!game.twoPlayer || this.id == "player2"){
459 			// Player 2 is mouse controlled
460 			if(game.leftButtonDown){ // left mouse button is pressed
461 				this.driveToPosition(game.frameDuration, game.mousePos,
462 					10, maxSpeed, turnRate, accel, decel);
463 			}else{
464 				this.velocity.accelerate(-decel * game.frameDuration);
465 			}
466 			if(game.rightButtonDown){ // right mouse is pressed
467 				// Turn to mouse pos and fire
468 				if(this.turnToDirection(game.frameDuration, game.mousePos, turnRate)){
469 					this.fire();
470 				}
471 			}
472 		}
473 	},
474 	step: function(p) {
475 		// Evaluate keyboard, mouse, and touch events
476 		this.applyControls();
477 		// --- Check for collisions
478 		var list = this.game.getObjectsByType("tank bandit");
479 		for(var i=0, l=list.length; i<l; i++) {
480 			var obj = list[i];
481 			if(!this.game.preCheckCollision(this, obj)){
482 				continue;
483 			}
484 			// Pre-check is exact enough for our purpose...
485 			obj.hitBy(this);
486 			this.hitBy(obj);
487 			break;
488 		}
489 	},
490 	render: function(ctx) {
491 		ctx.strokeStyle = this.color;
492 		ctx.strokePolygon2(this.pg, false);
493 //		ctx.strokeStyle = "magenta";
494 //		ctx.strokePolygon2(this.pgHull, false);
495 	},
496 	hitBy: function(obj) {
497 		this.game.explosionSound.play();
498 		// Score penalty for friendly fire
499 		var otherId = (obj.type == "bullet" ? obj.source.id : ""),
500 			isReal = this.isActivity("run");
501 		if(isReal && otherId == "player1"){
502 			this.game.score -= 100;
503 		}else if(isReal && otherId == "player2"){
504 			this.game.score2 -= 100;
505 		}
506 		this.hidden = true;
507 		this.game._assignTargets();
508 		this.hidden = false;
509 
510 		this.restart();
511 
512 //		this.later(this.game.gracePeriod, function(){
513 //			this.pos.set(this.homePos);
514 //			this.hidden = false;
515 //		});
516 	},
517 	/**Move outside screen and activate 'recover' mode (i.e. drive back to home)*/
518 	restart: function(){
519 		this.pos.set(this.recoverPos);
520 //		alert("restart: " + this.id + " home:" + this.homePos + ", recover:" + this.recoverPos);
521 		this.clipModeX = "none";
522 		this.setActivity("recover");
523 	},
524 	fire: function() {
525 		if(this.hidden || this.game.time - this.lastFire < this.fireRate){
526 			return;
527 		}
528 		this.lastFire = this.game.time;
529 		var bullet = new Bullet({
530 			source: this,
531 			pos: this.pos.copy(),
532 			velocity: LinaJS.polarToVec(this.orientation + 0.5 * Math.PI, 300),
533 			ttl: this.fireRange
534 			});
535 		this.game.addObject(bullet);
536 		this.game.gunSound.play();
537 	},
538 	// --- end of class
539 	__lastentry: undefined
540 });

544 var Bandit = Movable.extend({
545 	init: function(opts) {
546 		this._super("bandit", $.extend({
547 			clipModeX: "none",
548 			clipModeY: "none"
549 		}, opts));
550 		// Copy selected options as object attributes
551 		ArcadeJS.extendAttributes(this, opts, "score maxSpeed accel decel turnRate attackRange fireRate pg");
552 		this.target = null;
553 //		this.pgHull = this.pg.getConvexHull();
554 	},
555 	getBoundingCircle: function() {
556 		return new Circle2(this.pos, 9);
557 	},
558 	step: function(p) {
559 		var vDest,
560 			target = this.target,
561 			vTarget = this.pos.vectorTo(target.pos),
562 			minBuddyDist = 50,
563 			minCanisterDist = 50,
564 			epsTarget = 5, // WC units
565 			decel = this.opts.decel,
566 			self = this;
567 
568 		// Drag attached canister
569 		if(this.canister){
570 			this.canister.pos = this.pos.copy();
571 			if(this.velocity.length() > 15){
572 				this.canister.pos.translate(this.velocity.copy().revert().setLength(15));
573 			}
574 		}
575 
576 		// If a player is in reach, switch to attack mode
577 		this.attackMode = false;
578 		if( !this.isActivity("escape")){
579 			this.game.visitObjects(function(obj){
580 				vDest = self.pos.vectorTo(obj.pos);
581 				if(!obj.hidden && vDest.length() < self.attackRange){
582 					// temporarily override target
583 					target = obj;
584 					vTarget = vDest;
585 					self.attackMode = true;
586 					self.canister = null; // drop canister
587 					decel = 0; // I don't break for players
588 					return false;
589 				}
590 			}, "tank");
591 		}
592 //		// TODO: If a canister is in reach, go for it
593 //		this.game.visitObjects(function(obj){
594 //			vDest = obj.pos.vectorTo(self.pos);
595 //			if(vDest.length() < minCanisterDist){
596 ////				self.velocity.accelerate(vDest.limit(3), 100);
597 //				driveToPosition(self, this.frameDuration, obj.pos, 90*LinaJS.D2R, 50, 100, 50);
598 //			}
599 //		}, "canister");
600 
601 		// --- move to target
602 		var reached = this.driveToPosition(this.game.frameDuration, target.pos,
603 				epsTarget, this.maxSpeed, this.turnRate, this.accel, this.decel);
604 		if(reached){
605 			if(target.type == "canister"){
606 				// Reached a canister: attach and run home
607 				this.canister = target;
608 				var targetPos = new Point2(this.pos.x + LinaJS.random(-100, 100), -50);
609 				// flee to bottom if this is nearer
610 				if(this.pos.y > 240){
611 					targetPos.y = 530;
612 				}
613 				this.target = {type: "home", pos: targetPos};
614 				this.setActivity("escape");
615 			}else if(target.type == "home"){
616 				// Reached home: vanish
617 				this.canister.die();
618 				this.canister = null;
619 				this.target = null;
620 				this.die();
621 			}
622 		}
623 
624 		// --- avoid buddies
625 		this.game.visitObjects(function(obj){
626 			if(obj !== self ){
627 				vDest = obj.pos.vectorTo(self.pos);
628 				if(vDest.length() < minBuddyDist){
629 					self.velocity.accelerate(vDest.limit(3), this.maxSpeed);
630 				}
631 			}
632 		}, "bandit");
633 		// --- fire, if a player is in front of us
634 		if( this.attackMode //&& vTarget.length() < minFireDist
635 				&& LinaJS.angleDiff(vTarget.angle(), this.orientation - 90*LinaJS.D2R) < 25*LinaJS.D2R){
636 			this.fire();
